function[] = ex2num2(F, i)
    t = linspace(0,1,F);
    Y = 2*sin(165*pi*t) + 13*cos(6*pi*t) - 3*cos(80*pi*t);
    subplot(2,3,i);
    plot(t, Y);
    xlabel('temps (s)');
    title("Échantillonnage à " + F + "Hz");
end

function[r] = sumSine(n, t)
    r = 1/2;
    for i = 1:n
        e = sin(((2*i)+1)*t);
        e = e / ((2*i)+1);
        r = r + 2*e/pi;
    end
end

function[] = ex4num3(x, y)
    L = size(x, 1)/y*1000;
    Y = fft(x);
    P2 = abs(Y/(L));
    P1 = P2(1:L/2+1);
    P1(2:end-1) = 2*P1(2:end-1);
    f = y*(0:(L/2))/L;
    plot(f, P1);
end
